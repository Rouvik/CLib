91CLib_String CLib_Fs_dir(const char *path, const char *mask, uint8_t mode, const char *delim)
92{
93 WIN32_FIND_DATAA fdFile;
94 HANDLE hFind = NULL;
95
96 if (mask[0] == '\0') // make sure mask is not error type
97 {
98 mask = "*.*";
99 }
100
101 int pathLen = strlen(path);
102 int maskLen = strlen(mask);
103
104 char cpyPath[pathLen + maskLen + 1];
105 memcpy(cpyPath, path, pathLen);
106 memcpy(cpyPath + pathLen, mask, maskLen + 1);
107
108 CLib_String output = CLib_String_init("");
109
110 if ((hFind = FindFirstFileA(cpyPath, &fdFile)) == INVALID_HANDLE_VALUE)
111 {
112 fprintf(stderr, "[CBuilder FS Error] Failed to open path %s\n", cpyPath);
113 PRINT_LAST_ERROR
114 CLib_String_deinit(&output);
115 return (CLib_String){NULL, 0, 0};
116 }
117
118 do
119 {
120 if ((fdFile.cFileName[0] == '.') && (fdFile.cFileName[1] == '\0' || fdFile.cFileName[1] == '.'))
121 {
122 continue;
123 }
124
125 if ((mode & CLIB_FS_DIRMODE_FOLDERS) && (fdFile.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY))
126 {
127 CLib_String_concatCStr(&output, fdFile.cFileName);
128 }
129 else if (mode & CLIB_FS_DIRMODE_FILES && !(fdFile.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY))
130 {
131 CLib_String_concatCStr(&output, fdFile.cFileName);
132 }
133
134 CLib_String_concatCStr(&output, (char *)delim); // add the ending delimeter
135 } while (FindNextFileA(hFind, &fdFile));
136
137 FindClose(hFind); // cleanup
138
139 return output;
140}

142bool CLib_Fs_readFile(const char *path, CLib_String *str)
143{
144 char chunk[CLIB_BUF_CHUNK];
145 HANDLE hFile = CreateFileA(path, GENERIC_READ, FILE_SHARE_READ, NULL, OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, NULL);
146
147 if (hFile == INVALID_HANDLE_VALUE)
148 {
149 PRINT_LAST_ERROR
150 return true;
151 }
152
153 LARGE_INTEGER size;
154 if (!GetFileSizeEx(hFile, &size))
155 {
156 PRINT_LAST_ERROR
157 CloseHandle(hFile);
158 return true;
159 }
160
161 lldiv_t divValue = lldiv(size.QuadPart, CLIB_BUF_CHUNK);
162
163 for (int i = 0; i < divValue.quot; i++)
164 {
165 if (!ReadFile(hFile, chunk, CLIB_BUF_CHUNK, NULL, NULL))
166 {
167 PRINT_LAST_ERROR
168 CloseHandle(hFile);
169 return true;
170 }
171
172 CLib_String_concatCStr(str, chunk);
173 }
174
175 if (divValue.rem)
176 {
177 char remainder[divValue.rem];
178 if (!ReadFile(hFile, remainder, divValue.rem, NULL, NULL))
179 {
180 PRINT_LAST_ERROR
181 CloseHandle(hFile);
182 return true;
183 }
184
185 CLib_String_concatCStr(str, remainder);
186 }
187
188 CloseHandle(hFile);
189
190 return false; // success
191}

193bool CLib_Fs_writeFile(const char *path, CLib_String *contents, bool append)
194{
195 HANDLE hFile = CreateFileA(path, GENERIC_WRITE, FILE_SHARE_READ | FILE_SHARE_WRITE, NULL, (append ? OPEN_ALWAYS : CREATE_ALWAYS), FILE_ATTRIBUTE_NORMAL, NULL);
196 if (hFile == INVALID_HANDLE_VALUE)
197 {
198 PRINT_LAST_ERROR
199 return true;
200 }
201
202 if (append)
203 {
204 SetFilePointer(hFile, 0l, NULL, FILE_END);
205 }
206
207 DWORD bytesWritten;
208 WriteFile(hFile, contents->str, contents->len, &bytesWritten, NULL);
209 if (bytesWritten != contents->len)
210 {
211 PRINT_LAST_ERROR
212 CloseHandle(hFile);
213 return true;
214 }
215
216 CloseHandle(hFile);
217 return false;
218}

258CLib_String CLib_Fs_dir(const char *path, const char *mask, uint8_t mode, const char *delim)
259{
260 if (mask[0] == '/') // if first character is / then ignore
261 {
262 mask++;
263 }
264
265 struct dirent *dirEntry;
266 DIR *dir;
267 struct stat fileStatus;
268
269 dir = opendir(path);
270 if (!dir)
271 {
272 fprintf(stderr, "[CLib FS Error] Failed to open directory: %s\n", strerror(errno));
273 return (CLib_String){NULL, 0, 0}; // return nothing
274 }
275
276 CLib_String output = CLib_String_init("");
277
278 while ((dirEntry = readdir(dir)) != NULL)
279 {
280 // ignore . and .. entries
281 if ((dirEntry->d_name[0] == '.') && (dirEntry->d_name[1] == '\0' || dirEntry->d_name[1] == '.'))
282 {
283 continue;
284 }
285
286 char *dotPos = strchr(mask, '.') + 1;
287 if (!dotPos)
288 {
289 dotPos = (char *)mask + strlen(mask);
290 }
291
292 char *fileExt = strrchr(dirEntry->d_name, '.') + 1;
293 int fileNameLen = fileExt - dirEntry->d_name - 1;
294 if (fileExt == NULL + 1)
295 {
296 int dirLen = strlen(dirEntry->d_name);
297 fileExt = dirEntry->d_name + dirLen;
298 fileNameLen = dirLen;
299 }
300
301 char *fileName;
302 if (fileExt == dirEntry->d_name + 1)
303 {
304 fileName = "";
305 }
306 else
307 {
308 fileName = (char *)alloca(fileNameLen + 1);
309 memcpy(fileName, dirEntry->d_name, fileNameLen);
310 fileName[fileNameLen] = '\0';
311 }
312
313 int maskLen = strlen(mask);
314 if (mask[0] != '*' && strncmp(mask, fileName, fileNameLen - 1))
315 {
316 continue;
317 }
318
319 if (dotPos[0] != '*' && strcmp(dotPos, fileExt))
320 {
321 continue;
322 }
323
324 stat(dirEntry->d_name, &fileStatus);
325 if ((mode & CLIB_FS_DIRMODE_FOLDERS) && (fileStatus.st_mode & S_IFDIR))
326 {
327 CLib_String_concatCStr(&output, dirEntry->d_name);
328 }
329 else if ((mode & CLIB_FS_DIRMODE_FILES) && !(fileStatus.st_mode & S_IFDIR))
330 {
331 CLib_String_concatCStr(&output, dirEntry->d_name);
332 }
333
334 CLib_String_concatCStr(&output, (char *)delim);
335 }
336
337 closedir(dir); // cleanup
338
339 return output;
340}

342bool CLib_Fs_readFile(const char *path, CLib_String *str)
343{
344 char chunk[CLIB_BUF_CHUNK];
345 FILE *file = fopen(path, "rb");
346
347 if (!file)
348 {
349 fprintf(stderr, "[CLib FS Error] Failed to open file: %s\n", strerror(errno));
350 return true; // error
351 }
352
353 fseek(file, 0L, SEEK_END);
354 long long length = ftell(file);
355 rewind(file);
356
357 div_t divValue = div(length, CLIB_BUF_CHUNK);
358
359 for (int i = 0; i < divValue.quot; i++)
360 {
361 fread(chunk, CLIB_BUF_CHUNK, 1, file);
362 CLib_String_concatCStr(str, chunk);
363 }
364
365 if (divValue.rem)
366 {
367 char residual[divValue.rem];
368 fread(residual, divValue.rem, 1, file);
369 CLib_String_concatCStr(str, residual);
370 }
371
372 fclose(file);
373
374 return false; // success
375}

377bool CLib_Fs_writeFile(const char *path, CLib_String *contents, bool append)
378{
379 FILE *file = NULL;
380 if (append)
381 {
382 file = fopen(path, "a");
383 }
384 else
385 {
386 file = fopen(path, "w");
387 }
388
389 if (!file)
390 {
391 fprintf(stderr, "[CLib FS Error] Failed to open file: %s\n", strerror(errno));
392 return true;
393 }
394
395 int count = fwrite(contents->str, contents->len, 1, file);
396 if (!count)
397 {
398 fclose(file);
399 fprintf(stderr, "[CLib FS Error] Failed to write entire contents of string: %s\n", strerror(errno));
400 return false;
401 }
402
403 fclose(file);
404 return true;
405}
