58CLib_String CLib_String_init(const char *src)
59{
60 unsigned int count;
61 int len = strlen(src);
62 if (len == 0)
63 {
64 count = 1;
65 }
66 else
67 {
68 div_t divValue = div(len, CLIB_BUF_CHUNK);
69 count = divValue.quot;
70 if (divValue.rem) // allocate extra for spill
71 {
72 count++;
73 }
74 }
75
76 int buf_len = CLIB_BUF_CHUNK * count;
77 char *newMem = (char *)malloc(buf_len); // generate and copy the provided string to heap string
78 memcpy(newMem, src, len + 1);
79
80 return (CLib_String){
81 newMem,
82 len,
83 buf_len};
84}

176CLib_String *CLib_String_concat(CLib_String *str1, CLib_String *str2)
177{
178 int max_len = str1->len + str2->len + 2;
179 if (str1->buf_len < max_len) // not enough size, first allocate memory for concat [+2 for \0 chars]
180 {
181 div_t divValue = div(max_len, CLIB_BUF_CHUNK);
182 if (divValue.rem) // allocate extra chunk for spill
183 {
184 divValue.quot++;
185 }
186
187 str1->buf_len = CLIB_BUF_CHUNK * divValue.quot; // set new buf_len
188 str1->str = (char *)realloc(str1->str, str1->buf_len); // allocate more memory
189 }
190
191 memcpy((str1->str + str1->len), str2->str, str2->len + 1); // assumes str1 has enough space
192
193 str1->len += str2->len; // assign the new len
194
195 return str1; // return the str1 reference
196}

318void CLib_String_tokenizer(CLib_String *original, CLib_String *prevToken, const char *delim)
319{
320 if (prevToken->str == NULL) // first time tokenize
321 {
322 prevToken->str = original->str;
323 prevToken->len = 0;
324 prevToken->buf_len = 0; // assign the buffer length as 0 this proves that this string is a container string
325 }
326 else
327 {
328 prevToken->str += prevToken->len + 1; // update to new possible token location
329 }
330
331 char *ptr = prevToken->str;
332 while (strchr(delim, *ptr) != NULL && *ptr != '\0') // make sure ptr points to a valid string loc
333 {
334 ptr++;
335 }
336
337 prevToken->str = ptr; // assign the new pointer location to tokenize from
338 int len = 0;
339 while (strchr(delim, *ptr) == NULL && ptr - original->str < original->len)
340 {
341 len++;
342 ptr++;
343 }
344
345 prevToken->len = len;
346}

380CLib_String CLib_String_substr(CLib_String *str, unsigned int start, unsigned int length)
381{
382 if (start + length >= str->len)
383 {
384 return (CLib_String){NULL, 0, 0};
385 }
386
387 div_t divValue = div(length, CLIB_BUF_CHUNK);
388 if (divValue.rem)
389 {
390 divValue.quot++;
391 }
392
393 CLib_String substr;
394 substr.buf_len = divValue.quot * CLIB_BUF_CHUNK;
395 substr.str = (char *)malloc(substr.buf_len);
396 memcpy(substr.str, str->str + start, length);
397 *(substr.str + length) = '\0';
398 substr.len = length;
399
400 return substr;
401}
