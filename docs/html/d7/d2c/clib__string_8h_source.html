49CLib_String CLib_String_init(const char *src)
50{
51 unsigned int count;
52 int len = strlen(src);
53 if (len == 0)
54 {
55 count = 1;
56 }
57 else
58 {
59 div_t divValue = div(len, CLIB_BUF_CHUNK);
60 count = divValue.quot;
61 if (divValue.rem) // allocate extra for spill
62 {
63 count++;
64 }
65 }
66
67 int buf_len = CLIB_BUF_CHUNK * count;
68 char *newMem = (char *)malloc(buf_len); // generate and copy the provided string to heap string
69 memcpy(newMem, src, len + 1);
70
71 return (CLib_String){
72 newMem,
73 len,
74 buf_len};
75}

168CLib_String *CLib_String_concat(CLib_String *str1, CLib_String *str2)
169{
170 int max_len = str1->len + str2->len + 2;
171 if (str1->buf_len < max_len) // not enough size, first allocate memory for concat [+2 for \0 chars]
172 {
173 div_t divValue = div(max_len, CLIB_BUF_CHUNK);
174 if (divValue.rem) // allocate extra chunk for spill
175 {
176 divValue.quot++;
177 }
178
179 str1->buf_len = CLIB_BUF_CHUNK * divValue.quot; // set new buf_len
180 str1->str = (char *)realloc(str1->str, str1->buf_len); // allocate more memory
181 }
182
183 memcpy((str1->str + str1->len), str2->str, str2->len + 1); // assumes str1 has enough space
184
185 str1->len += str2->len; // assign the new len
186
187 return str1; // return the str1 reference
188}

310void CLib_String_tokenizer(CLib_String *original, CLib_String *prevToken, const char *delim)
311{
312 if (prevToken->str == NULL) // first time tokenize
313 {
314 prevToken->str = original->str;
315 prevToken->len = 0;
316 prevToken->buf_len = 0; // assign the buffer length as 0 this proves that this string is a container string
317 }
318 else
319 {
320 prevToken->str += prevToken->len + 1; // update to new possible token location
321 }
322
323 char *ptr = prevToken->str;
324 while (strchr(delim, *ptr) != NULL && *ptr != '\0') // make sure ptr points to a valid string loc
325 {
326 ptr++;
327 }
328
329 prevToken->str = ptr; // assign the new pointer location to tokenize from
330 int len = 0;
331 while (strchr(delim, *ptr) == NULL && ptr - original->str < original->len)
332 {
333 len++;
334 ptr++;
335 }
336
337 prevToken->len = len;
338}

371CLib_String CLib_String_substr(CLib_String *str, unsigned int start, unsigned int length)
372{
373 if (start + length >= str->len)
374 {
375 return (CLib_String){NULL, 0, 0};
376 }
377
378 div_t divValue = div(length, CLIB_BUF_CHUNK);
379 if (divValue.rem)
380 {
381 divValue.quot++;
382 }
383
384 CLib_String substr;
385 substr.buf_len = divValue.quot * CLIB_BUF_CHUNK;
386 substr.str = (char *)malloc(substr.buf_len);
387 memcpy(substr.str, str->str + start, length);
388 *(substr.str + length) = '\0';
389 substr.len = length;
390
391 return substr;
392}
